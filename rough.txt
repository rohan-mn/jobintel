# UTF-8 without BOM keeps repository files consistent across Windows and Linux containers.
$Utf8NoBom = New-Object System.Text.UTF8Encoding($false)

function Write-Utf8File {
    param(
        [Parameter(Mandatory = $true)]
        [string]$Path,

        [Parameter(Mandatory = $true)]
        [AllowEmptyString()]
        [string]$Content
    )

    $ParentDirectory = Split-Path -Parent $Path

    if ($ParentDirectory -and -not (Test-Path $ParentDirectory)) {
        New-Item -ItemType Directory -Path $ParentDirectory -Force | Out-Null
    }

    [System.IO.File]::WriteAllText(
        (Join-Path (Get-Location) $Path),
        $Content.TrimStart(),
        $Utf8NoBom
    )
}

# -------------------------------------------------------------------
# Function for creating workspace package manifests
# -------------------------------------------------------------------

function New-WorkspacePackage {
    param(
        [Parameter(Mandatory = $true)]
        [string]$Directory,

        [Parameter(Mandatory = $true)]
        [string]$Name,

        [Parameter(Mandatory = $true)]
        [string]$Description
    )

    $PackageJson = @{
        name        = $Name
        version     = "0.0.0"
        private     = $true
        description = $Description
        type        = "module"
    } | ConvertTo-Json -Depth 10

    Write-Utf8File "$Directory/package.json" $PackageJson
    Write-Utf8File "$Directory/src/.gitkeep" ""
}
